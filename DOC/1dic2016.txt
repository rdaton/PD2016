data Complex = C Double Double
	deriving Eq
	
instance num complex where 
	C a b + C c d = C(a+c) (b+d)
	C a b - C c d = C(a-c) (b-d)
	C a b * C c d = C(a-c) (b-d)
	
--el tipo de show ya est√° dado	
instance show complex where
	show (C x y )
